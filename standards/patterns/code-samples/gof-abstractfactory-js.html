function StandardEmployee() {
    this.usefulfunction = function () {
        log.add("I am a standard employee.");
    };
}
function StandardConsultant() {
    this.usefulfunction = function () {
        log.add("I am a standard consultant.");
    };
}
function StandardFactory() {
    this.createEmployee = function() {
        return new StandardEmployee();
    };
    this.createConsultant = function() {
        return new StandardConsultant();
    };
}

function PreferredEmployee() {
    this.usefulfunction = function () {
        log.add("I am a preferred employee.");
    };
}
function PreferredConsultant() {
    this.usefulfunction = function () {
        log.add("I am a preferred consultant.");
    };
}
function PreferredFactory() {
    this.createEmployee = function() {
        return new PreferredEmployee();
    };
    this.createConsultant = function() {
        return new PreferredConsultant();
    };
}
    
// log helper
var log = (function () {
    var log = "";
    return {
        add: function (msg) { log += msg + "\n"; },
        show: function () { alert(log); log = ""; }
    }
})();
    
function run() {
    var persons = [];
    var standardFactory = new StandardFactory();
    var preferredFactory = new PreferredFactory();
    
    persons.push(standardFactory.createEmployee());
    persons.push(standardFactory.createConsultant());
    persons.push(preferredFactory.createEmployee());
    persons.push(preferredFactory.createConsultant());
    
    for (var i = 0, len = persons.length; i < len; i++) {
        persons[i].usefulfunction();
    }
    
    log.show();
}
