// the Facade class
class Mortgage {
    private Bank _bank = new Bank();
    private Credit _credit = new Credit();
    private Loan _loan = new Loan();
    private boolean _eligible = true;
    
    public boolean isEligible(Customer customer, int amount) {
        if(!_bank.hasSufficientSavings(customer, amount)) {
            _eligible = false;
        } else if(!_loan.hasNoBadLoans(customer)) {
            _eligible = false;
        } else if(!_credit.hasGoodCredit(customer)) {
            _eligible = false;
        }
        return _eligible;
    }
}

public class TestFacade {
    public static void main(String[] args) {
        // create Facade
        Mortgage mortgage = new Mortgage();
        // check eligibility for customer
        Customer customer = new Customer("Jerry Shaughnessy");
        boolean eligible = mortgage.isEligible(customer, 125000);
        
        System.out.println("\n" + customer.Name + " has been " + (eligible ? "Approved" : "Rejected"));
    }
}
