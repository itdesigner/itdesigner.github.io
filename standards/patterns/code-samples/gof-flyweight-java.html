import java.util.Map;
import java.util.HashMap;

public class Demo {
    public static void main(String[] args) {
        /* Build a document with text */
        String document = "AZAZZBBZB";
        char[] chars = document.toCharArray();
        /* extrinsic state */
        int pointSize = 10;
        /* character creation */
        CharacterFactory factory = new CharacterFactory();
        // For each character use a flyweight object
        for (char c : chars) 
        {
            pointSize++;
            CharacterBase character = factory.getCharacter(c);
            character.display(pointSize);
        }
    }
}

/* Flyweight Factory class */
class CharacterFactory {
    private final Map<Character, CharacterBase> _characters;
    public CharacterFactory() {
        _characters = new HashMap<Character, CharacterBase>();
    }

    CharacterBase getCharacter(char key) {
        /* lazy initialization */
        CharacterBase character = _characters.get(key);
        if (character == null) {
            System.out.println(key + " NOT found in cache");
            switch (key) {
                case 'A':
                    character = new CharacterA();
                    _characters.put(key, character);
                    break;
                case 'B':
                    character = new CharacterB();
                    _characters.put(key, character);
                    break;
                /* ... */
                case 'Z':
                    character = new CharacterZ();
                    _characters.put(key, character);
                    break;
                default:
                    break;
            }
        } else {
            System.out.println(key + " found in cache");
        }
        return character;
    }
}

/* Flyweight class */
abstract class CharacterBase {
    protected char symbol;
    protected int width;
    protected int height;
    protected int pointSize;
    protected String font;
    protected abstract void display(int pointSize);
}

/* ConcreteFlyweight class */
class CharacterA extends CharacterBase {
    // Constructor
    public CharacterA() {
        this.symbol = 'A';
        this.height = 100;
        this.width = 120;
        this.font = "Arial";
    }
    @Override
    public void display(int pointSize) {
        System.out.println(this.symbol + " (pointsize: " + pointSize + " font: " + this.font + ")");
    }
}

/* ConcreteFlyweight class */
class CharacterB extends CharacterBase {
    // Constructor
    public CharacterB() {
        this.symbol = 'B';
        this.height = 100;
        this.width = 120;
        this.font = "Arial";
    }
    @Override
    public void display(int pointSize) {
        System.out.println(this.symbol + " (pointsize: " + pointSize + " font: " + this.font + ")");
    }
}
