    }

var Credit = (function () {
    function Credit() {
    }
    Credit.prototype.hasGoodCredit = function (customer) {
        log.add("checking credit for " + customer.Name);
        // complex credit check logic
        return true;
    };

}());

// subsystem
var Customer =  (function () {
    function Customer(name) {
        this.Name = name;
    }
    return Customer;
}());

// facade
var Mortgage = (function () {
    function Mortgage() {
        this.bank = new Bank();
        this.loan = new Loan();
        this.credit = new Credit();
    }
    Mortgage.prototype.isEligible = function (customer, amount) {
        log.add(customer.Name + " applying for loan");
        var eligible = true;
        // check subsystems
        if (this.bank.hasSufficientSavings(customer, amount) === false) {
            eligible = false;
        }
        else if (this.loan.hasNoBadLoans(customer) === false) {
            eligible = false;
        }
        else if (this.credit.hasGoodCredit(customer) === false) {
            eligible = false;
        }
        return eligible;
    };
    return Mortgage;
}());

// log helper class
var log = (function () {
    function log() {
    }
    log.add = function (message) {
        log._log += message + '\n';
    };
    log.show = function () {
        return log._log;
        log._log = '';
    };
    log._log = '';
    return log;
}());

// create facade
var mortgage = new Mortgage();
var customer = new Customer("Jerry Shaughnessy");
// evaluate
var result = mortgage.isEligible(customer, 129000);
if (result) {
    log.add(customer.Name + " has been Approved for a loan");
}
else {
    log.add(customer.Name + " has been Rejected for a loan");
}
document.body.innerHTML = log.show();
