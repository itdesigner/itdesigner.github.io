public class MyClass {
    public static void main(String args[]) {
        // create a yen
        ICurrency currency = CurrencyFactory.createCurrency("yen");
        currency.describe();

        // create a euro
        currency = CurrencyFactory.createCurrency("euro");
        currency.describe();

        // create a dollar
        currency = CurrencyFactory.createCurrency("dollar");
        currency.describe();
    }
}
interface ICurrency
{
    public void describe();
}
/// <summary>
/// A 'ConcreteCurrency' class
/// </summary>
class YenCurrency implements ICurrency
{
    public String type="";
    public double rate=0.00;
    public String country="";
    public YenCurrency()
    {
        this.rate = 113.40;
        this.country = "Japan";
        this.type = "yen";
    }
    public void describe()
    {
        System.out.println(this.type + ": from " + this.country + " currently at " + this.rate);
    }
}
/// <summary>
/// A 'ConcreteCurrency' class
/// </summary>
class EuroCurrency implements ICurrency
{
    public String type="";
    public double rate=0.00;
    public String country="";
    public EuroCurrency()
    {
        this.rate = 0.86;
        this.country = "European Union";
        this.type = "euro";
    }
    public void describe()
    {
        System.out.println(this.type + ": from " + this.country + " currently at " + this.rate);
    }
}
/// <summary>
/// A 'ConcreteCurrency' class
/// </summary>
class DollarCurrency implements ICurrency
{
    public String type="";
    public double rate=0.00;
    public String country="";
    public DollarCurrency()
    {
        this.rate = 1.00;
        this.country = "United States";
        this.type = "dollar";
    }
    public void describe()
    {
        System.out.println(this.type + ": from " + this.country + " currently at " + this.rate);
    }
}
class CurrencyFactory
{
    public static ICurrency createCurrency(String criteria)
    {
        if ( criteria.equals("yen") )
            return new YenCurrency();
        else if ( criteria.equals("euro") )
            return new EuroCurrency();
        else if ( criteria.equals("dollar") )
            return new DollarCurrency();
        
        return null;
    }
}
