function EmployeePrototype(proto) {
    this.proto = proto;

    this.clone = function () {
        var dev = new Developer();

        dev.name = proto.name;
        dev.role = proto.role;
        dev.preferredLanguage = proto.preferredLanguage;

        return dev;
    };
}

function Developer(name, role, preferredLanguage) {

    this.name = name;
    this.role = role;
    this.preferredLanguage = preferredLanguage;

    this.getDetails = function () {
        return ("name: " + this.name + " "  +
                ", role: " + this.role + " - language: " + this.preferredLanguage);
    };
}
// log helper class
class log {
    private static _log: string = '';
    static add(message: string): void {
        log._log += message + '\n';
    }
    static show(): string {
        return log._log;
        log._log = '';
    }
}

var proto = new Developer("Michael", "Team Lead", "JS");
log.add(proto.getDetails());
var prototype = new EmployeePrototype(proto);

var customer = prototype.clone();
customer.name = "Jose";
log.add(customer.getDetails());

log.show();
