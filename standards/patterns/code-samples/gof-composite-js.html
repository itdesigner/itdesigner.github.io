var GroupConfig = (function () {
    function GroupConfig(name) {
        this._children = new Array();
        this._settings = "";
        this._name = name;
    }
    Object.defineProperty(GroupConfig.prototype, "Name", {
        get: function () { return this._name; },
        enumerable: true,
        configurable: true
    });
    GroupConfig.prototype.addSettings = function (setting) {
        this._settings = this._settings + setting + ";";
    };
    GroupConfig.prototype.getSettings = function (parentSettings, depth) {
        var s = parentSettings + this._settings;
        var p = "";
        for (var i = 0; i < depth; i++) {
            p = p + "-";
        }
        log.add(p + this._name + " : " + s);
        // Recursively display child nodes
        this._children.forEach(function (child) {
            child.getSettings(s, depth + 1);
        });
    };
    GroupConfig.prototype.add = function (c) {
        this._children.push(c);
    };
    GroupConfig.prototype.remove = function (i) {
        if (this._children.length <= i) {
            throw new Error("index out of bound!");
        }
        this._children.splice(i, 1);
    };
    return GroupConfig;
}());
