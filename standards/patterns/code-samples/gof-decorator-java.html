public class DecoratorPatternDemo {
    public static void main(String[] args) {
        Component comp = new ConcreteComponent();
        Component d1 = new TimingDecorator(comp);
        Component d2 = new ConcreteDecoratorB(d1);
        d2.operation("abcd");
    }
}

interface Component {
    void operation(String args);
}

class ConcreteComponent implements Component {
    @Override
    public void operation(String args) {
        // something that takes time to measure
        int k = 0;
        for(int i = 0; i<200000; i++)
        {
            k += i;
        }
        System.out.println("ConcreteComponent.Operation(" + args + ") - DONE");
    }
}

abstract class BaseDecorator implements Component {
    protected Component component;
    public BaseDecorator(Component component){
        this.component = component;
    }

    public void operation(String args){
        component.operation(args);
    }	
}

class TimingDecorator extends BaseDecorator {
    public TimingDecorator(Component component) {
        super(component);		
    }
    @Override
    public void operation(String args) {
        System.out.println("TimingDecorator.pre-Operation()");
        long startTime = System.nanoTime();
        component.operation(args);	       
        long endTime = System.nanoTime();
        long totalMilliseconds = ((endTime-startTime)/1000000);
        System.out.println("TimingDecorator.post-Operation(): " + totalMilliseconds + "ms");
    }
}

class ConcreteDecoratorB extends BaseDecorator {
    public ConcreteDecoratorB(Component component) {
        super(component);		
    }

    @Override
    public void operation(String args) {
        System.out.println("ConcreteDecoratorB pre-Operation()");
        component.operation(args);	       
        System.out.println("ConcreteDecoratorB post-Operation()");
    }
}
