export interface IAbstractEmployee {
    usefulfunction(): string;
}
export interface IAbstractConsultant {
    usefulfunction(): string;
}

export interface IAbstractFactory {
    createEmployee(param?: any) : IAbstractEmployee;
    createConsultant() : IAbstractConsultant;
}


export class StandardEmployee implements IAbstractEmployee {
    usefulfunction = () => {
        return "This is methodA of StandardEmployee";
    }
}
export class StandardConsultant implements IAbstractConsultant {
    usefulfunction = () => {
        return "This is methodB of StandardConsultant";
    }
}

export class PreferredEmployee implements IAbstractEmployee {
    usefulfunction = () => {
        return "This is methodA of PreferredEmployee";
    }
}
export class PreferredConsultant implements IAbstractConsultant {
    usefulfunction = () => {
        return "This is methodB of PreferredConsultant";
    }
}


export class StandardFactory implements AbstractFactory {
    createEmployee(param?: any) : IAbstractEmployee {
        return new StandardEmployee();
    }

    createConsultant(param?: any) : IAbstractConsultant {
        return new StandardConsultant();
    }
}
export class PreferredFactory implements AbstractFactory {
    createEmployee(param?: any) : IAbstractEmployee {
        return new PreferredEmployee();
    }

    createConsultant(param?: any) : IAbstractConsultant {
        return new PreferredConsultant();
    }
}

var persons = [];
var standardFactory: IAbstractFactory  = new StandardFactory();
var preferredFactory: IAbstractFactory = new PreferredFactory();


persons.push(standardFactory.createEmployee());
persons.push(standardFactory.createConsultant());
persons.push(preferredFactory.createEmployee());
persons.push(preferredFactory.createConsultant());

for (var i = 0, len = persons.length; i < len; i++) {
    persons[i].usefulfunction();
}
