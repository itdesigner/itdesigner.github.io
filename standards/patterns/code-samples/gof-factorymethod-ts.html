interface ICurrency {
    name: string;
    rate: number;
    country: string;
    describe(): void;
    
}
class YenCurrency implements ICurrency {
    name: string;
    rate: number;
    country: string;
    constructor() {
        this.rate = 113.40;
        this.country = 'Japan';
        this.name = 'yen';
    }
    describe(): void {
        log.add(this.name + ': from ' + this.country + ' currently at ' + this.rate);
    }
}
class EuroCurrency implements ICurrency {
    name: string;
    rate: number;
    country: string;
    constructor() {
        this.rate = 0.86;
        this.country = 'European Union';
        this.name = 'euro';
    }
    describe(): void {
        log.add(this.name + ': from ' + this.country + ' currently at ' + this.rate);
    }
}
class DollarCurrency implements ICurrency {
    name: string;
    rate: number;
    country: string;
    constructor() {
        this.rate = 1.00;
        this.country = 'United States';
        this.name = 'dollar';
    }
    describe(): void {
        log.add(this.name + ': from ' + this.country + ' currently at ' + this.rate);
    }
}
class Factory {
    public static createCurrency(type: string): ICurrency {
        var _currency: ICurrency;
        if (type === 'yen') {
            _currency = new YenCurrency();
        } else if (type === 'euro') {
            _currency = new EuroCurrency();
        } else if (type === 'dollar') {
            _currency = new DollarCurrency();
        }
        return _currency;
    }
}
// log helper class
class log {
    private static _log: string = '';
    static add(message: string): void {
        log._log += message + '\n';
    }
    static show(): string {
        return log._log;
        log._log = '';
    }
}

var currencies: ICurrency[] =[];
currencies.push(Factory.createCurrency('yen'));
currencies.push(Factory.createCurrency("euro"));
currencies.push(Factory.createCurrency("dollar"));
    
for (var i = 0, len = currencies.length; i < len; i++) {
    currencies[i].describe();
}

document.body.innerHTML = log.show();
