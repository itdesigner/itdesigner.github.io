using System;
                
/// <summary>
/// Factory Pattern Demo
/// </summary>
public class Program
{
    public static void Main(string[] args)
    {
        CurrencyFactory factory = new ConcreteCurrencyFactory();

        ICurrency yen = factory.createCurrency("yen");
        yen.describe();
        ICurrency euro = factory.createCurrency("euro");
        euro.describe();
        ICurrency dollar = factory.createCurrency("dollar");
        dollar.describe();
    }
}

/// <summary>
/// The 'Currency' interface
/// </summary>
public interface ICurrency
{
    string type { get; set; }
    double rate { get; set; }
    string country { get; set; }
    void describe();
}

/// <summary>
/// A 'ConcreteCurrency' class
/// </summary>
public class YenCurrency : ICurrency
{
    public string type { get; set; }
    public double rate { get; set; }
    public string country { get; set; }
    public YenCurrency()
    {
        this.rate = 113.40;
        this.country = "Japan";
        this.type = "yen";
    }
    public void describe()
    {
        Console.WriteLine(this.type + ": from " + this.country + " currently at " + this.rate);
    }
}

/// <summary>
/// A 'ConcreteCurrency' class
/// </summary>
public class EuroCurrency : ICurrency
{
    public string type { get; set; }
    public double rate { get; set; }
    public string country { get; set; }
    public EuroCurrency()
    {
        this.rate = 0.86;
        this.country = "European Union";
        this.type = "euro";
    }
    public void describe()
    {
        Console.WriteLine(this.type + ": from " + this.country + " currently at " + this.rate);
    }
}

/// <summary>
/// A 'ConcreteCurrency' class
/// </summary>
public class DollarCurrency : ICurrency
{
    public string type { get; set; }
    public double rate { get; set; }
    public string country { get; set; }
    public DollarCurrency()
    {
        this.rate = 1.00;
        this.country = "United States";
        this.type = "dollar";
    }
    public void describe()
    {
        Console.WriteLine(this.type + ": from " + this.country + " currently at " + this.rate);
    }
}

/// <summary>
/// The Currency Factory Abstract Class
/// </summary>
public abstract class CurrencyFactory
{
    public abstract ICurrency createCurrency(string currencyType);

}

/// <summary>
/// A 'ConcreteCreator' class
/// </summary>
public class ConcreteCurrencyFactory : CurrencyFactory
{
    public override ICurrency createCurrency(string currencyType)
    {
        switch (currencyType)
        {
            case "yen":
                return new YenCurrency();
            case "euro":
                return new EuroCurrency();
            case "dollar":
                return new DollarCurrency();
            default:
                throw new ApplicationException(string.Format("Currency '{0}' cannot be created", currencyType));
        }
    }
}
