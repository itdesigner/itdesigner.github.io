interface IEmployee {
        clone(): IEmployee;
        getDetails(): string;
    }

class Developer implements IEmployee {
      public WordsPerMinute: number;
      public Name: string;
      public Role: string;
      public PreferredLanguage: string;

      constructor(obj?: any) {
          if(obj !=null) {
              this.WordsPerMinute = obj.WordsPerMinute;
              this.Name = obj.Name;
              this.Role = obj.Role;
              this.PreferredLanguage = obj.PreferredLanguage;
          }
        }
        public clone(): Developer {
            return (new Developer(this));
        }

        public getDetails(): string {
            return " DEV: " + this.Name + " - " + this.Role + " " + this.PreferredLanguage;;
        }
    }

class Admin implements IEmployee {
      public WordsPerMinute: number;
      public Name: string;
      public Role: string;

      constructor(obj?: any) {
          if(obj !=null) {
              this.WordsPerMinute = obj.WordsPerMinute;
              this.Name = obj.Name;
              this.Role = obj.Role;
          }
        }
        public clone(): Admin {
            return (new Admin(this));
        }

        public getDetails(): string {
            return "ADMIN: " + this.Name + " - " + this.Role + " " + this.WordsPerMinute + "wpm";
        }
    }
// log helper class
class log {
    private static _log: string = '';
    static add(message: string): void {
        log._log += message + '\n';
    }
    static show(): string {
        return log._log;
        log._log = '';
    }
}

var admin: Admin = new Admin();
admin.Name = "Jane";
admin.Role = "Administrative Assistant";
admin.WordsPerMinute = 140;
var anotherAdmin: Admin = admin.clone();
anotherAdmin.Name = "Robert";
log.add(admin.getDetails());
log.add(anotherAdmin.getDetails());
