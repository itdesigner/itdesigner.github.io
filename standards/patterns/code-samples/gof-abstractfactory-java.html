    } // Implement the code here
}

abstract class AbstractFactory{
    abstract IAbstractEmployee createEmployee();
    abstract IAbstractConsultant createConsultant();
}

class StandardFactory extends AbstractFactory{
    IAbstractEmployee createEmployee(){
        return new StandardEmployee("StandardEmployee");
    }
    IAbstractConsultant createConsultant(){
        return new StandardConsultant("StandardConsultant");
    }
}

class PreferredFactory extends AbstractFactory{
    IAbstractEmployee createEmployee(){
        return new PreferredEmployee("PreferredEmployee");
    }
    IAbstractConsultant createConsultant(){
        return new PreferredConsultant("PreferredConsultant");
    }
}

//Factory creator - an indirect way of instantiating the factories
class FactoryMaker{
    private static AbstractFactory pf=null;
    static AbstractFactory getFactory(String choice){
        if(choice.equals("s")){
            pf=new StandardFactory();
        }else if(choice.equals("p")){
            pf=new PreferredFactory();
        } return pf;
    }
}
