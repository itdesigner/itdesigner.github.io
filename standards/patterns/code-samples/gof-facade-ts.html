// subsystem
class Bank {
    public hasSufficientSavings(customer: Customer, amount: number): boolean {
        log.add("checking savings balance for " + customer.Name);
        return true;
    }
}

// subsystem
class Loan {
    public hasNoBadLoans(customer: Customer): boolean {
        log.add("checking loan history for " + customer.Name);
        // complex history check logic
        return true;
    }
}

// subsystem
class Credit {
    public hasGoodCredit(customer: Customer): boolean {
        log.add("checking credit for " + customer.Name);
        // complex credit check logic
        return true;
    }
}

// subsystem
class Customer {
    Name: string;
    constructor(name: string): void {
        this.Name = name;
    }
}

class Mortgage {
    private bank: Bank;
    private loan: Loan;
    private credit: Credit;
        constructor() {
        this.bank = new Bank();
        this.loan = new Loan();
            this.credit  = new Credit();
    }

    public isEligible(customer: Customer, amount: number): boolean {
        log.add(customer.Name + " applying for loan");
        var eligible: boolean = true;

        // check subsystems
        if (this.bank.hasSufficientSavings(customer, amount) === false) {
            eligible = false;
        } else if(this.loan.hasNoBadLoans(customer) === false) {
            eligible = false;
        } else if(this.credit.hasGoodCredit(customer) === false) {
            eligible=false;
        }
        return eligible;
    }
}

// log helper class
class log {
    private static _log: string = '';
    static add(message: string): void {
        log._log += message + '\n';
    }
    static show(): string {
        return log._log;
        log._log = '';
    }
}

// create facade
var mortgage: Mortgage = new Mortgage();
var customer: Customer = new Customer("Jerry Shaughnessy");
// evaluate
var result: boolean = mortgage.isEligible(customer, 129000);
if(result) {
        log.add(customer.Name + " has been Approved for a loan");
} else {
        log.add(customer.Name + " has been Rejected for a loan");
}


document.body.innerHTML = log.show();
