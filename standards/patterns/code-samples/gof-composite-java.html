import java.util.ArrayList;
import java.util.Iterator;
import java.util.List;

/// <summary>
/// MainApp startup class for Structural
/// Composite Design Pattern.
/// </summary>
public class MainApp
{
    /// <summary>
    /// Entry point into console application.
    /// </summary>
    public static void main (String[] args) 
    {
        // Create a tree structure
        ConfigComponent root = new GroupConfig("root");
        root.AddSettings("visible=true");
        ConfigComponent m1 = new MachineConfig("Machine R-1");
        m1.AddSettings("test=true");
        ConfigComponent m2 = new MachineConfig("Machine R-2");
        m2.AddSettings("test=false");
        root.Add(m1);
        root.Add(m2);

        ConfigComponent comp = new GroupConfig("Group A");
        comp.AddSettings("theme=blue");
        ConfigComponent m3 = new MachineConfig("Machine A-3");
        m3.AddSettings("network=public");
        ConfigComponent m4 = new MachineConfig("Machine A-4");
        m4.AddSettings("network=private");
        comp.Add(m3);
        comp.Add(m4);

        root.Add(comp);
        ConfigComponent m5 = new MachineConfig("Machine R-5");
        m5.AddSettings("type=internal");
        root.Add(m5);

        // Add and remove a leaf
        MachineConfig leaf = new MachineConfig("Machine R-6");
        root.Add(leaf);
        root.Remove(leaf);

        // Recursively get settings
        root.GetSettings("", 1);

    }
}

/// <summary>
/// The 'Component' abstract class
/// </summary>
abstract class ConfigComponent
{
    public String name;
    // Constructor
    public ConfigComponent(String name)
    {
        this.name = name;
    }
    public abstract void Add(ConfigComponent c);
    public abstract void Remove(ConfigComponent c);
    public abstract void AddSettings(String setting);
    public abstract void GetSettings(String parentSettings, int depth);
}

/// <summary>
/// The 'Composite' class
/// </summary>
class GroupConfig extends ConfigComponent
{
    private String _settings = "";
    List<ConfigComponent> _children = new ArrayList<ConfigComponent>();
    // Constructor
    public GroupConfig(String name)
    {
        super(name);
    }
    @Override
    public void Add(ConfigComponent component)
    {
        _children.add(component);
    }
    @Override
    public void Remove(ConfigComponent component)
    {
        _children.remove(component);
    }
    @Override
    public void AddSettings(String setting)
    {
        _settings += setting + ";";
    }
    @Override
    public void GetSettings(String parentSettings, int depth)
    {
        String s = parentSettings + this._settings;
        String p = "";
        for(int i=0; i<depth; i++) {
            p += "-";
        }
        System.out.println(p + this.name + " : " + s);
        // Recursively display child nodes
        for (ConfigComponent component : _children)
        {
            component.GetSettings(s, depth + 1);
        }
    }
}

/// <summary>
/// The 'Leaf' class
/// </summary>
class MachineConfig extends ConfigComponent
{
    private String _settings = "";
    // Constructor
    public MachineConfig(String name)
    {
        super(name);
    }
    @Override
    public void Add(ConfigComponent c)
    {
        System.out.println("Cannot add to a leaf");
    }
    @Override
    public void Remove(ConfigComponent c)
    {
        System.out.println("Cannot remove from a leaf");
    }
    @Override
    public void AddSettings(String setting)
    {
        this._settings += setting + ";";;
    }
    @Override
    public void GetSettings(String parentSettings, int depth)
    {
        String s = parentSettings + this._settings;
        String p = "";
        for(int i=0; i<depth; i++) {
            p += "-";
        }
        System.out.println(p + this.name + " : " + s);
    }
}
