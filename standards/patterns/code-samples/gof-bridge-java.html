/**
* abstraction in bridge pattern
* */
abstract class WebApplication {
    public ITheme Theme;
    public String Title;
    public String SiteName;
    protected WebApplication() { }
    abstract public void buildPage();
}

/**
 * Refine abstraction 1 in bridge pattern
 */
class BlogApp extends WebApplication {
    public BlogApp() {
        super();
    }
    public String Content;
    @Override
    public void buildPage() {
        String pageBody =  this.Title + " " + this.SiteName + " - " + this.Content;
        Theme.ApplyTheme(pageBody);;
    }
}

/**
* Refine abstraction 2 in bridge pattern
*/
class NewsApp extends WebApplication {
    public NewsApp() {
        super();
    }
    public String Story;
    @Override
    public void buildPage() {
        String pageBody =  this.Title + " " + this.SiteName + " - " + this.Story;
        Theme.ApplyTheme(pageBody);;
    }
}

/**
* Refine abstraction 3 in bridge pattern
*/
class StoreApp extends WebApplication {
    public StoreApp() {
        super();
    }
    public String Items;
    @Override
    public void buildPage() {
        String pageBody =  this.Title + " " + this.SiteName + " - " + this.Items;
        Theme.ApplyTheme(pageBody);;
    }
}

/**
* Implementor for bridge pattern
* */
interface ITheme {
    abstract public void ApplyTheme(String content);
}

/**
* Concrete implementation 1 for bridge pattern
* */
class DarkTheme implements ITheme {
    @Override
    public void ApplyTheme(String content) {
        System.out.print(content + " ...applied dark theme\n");
    }
}

/**
* Concrete implementation 2 for bridge pattern
* */
class LightTheme implements ITheme {
    @Override
    public void ApplyTheme(String content) {
        System.out.println(content + " ...applied light theme\n");
    }
}

/*
* Demonstration of bridge design pattern
*/
public class Program {
    public static void main(String[] args) {
        ITheme dark = new DarkTheme();
        ITheme light = new LightTheme();

        WebApplication app = new BlogApp();
        app.Title = "Blog-o-rama";
        app.SiteName = "TestSite";
        ((BlogApp)app).Content = "My blog entry";
        app.Theme = dark;
        app.buildPage();
        app.Theme = light;
        app.buildPage();

        WebApplication news = new NewsApp();
        news.Title = "News-o-rama";
        news.SiteName = "TestSite";
        ((NewsApp)news).Story = "breaking news story";
        news.Theme = light;
        news.buildPage();
    }
}
