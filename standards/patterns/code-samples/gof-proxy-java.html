// the ISubject interface
interface ISubject {
    public void draw();
}

// the RealSubject class
class BrowserImage implements ISubject {

    private String filename = null;
    private byte[] imageData = null;
    //Constructor
    public BrowserImage(final String filename) {
        this.filename = filename;
        loadImageFromDisk();
        System.out.println("Creating BrowserImage");
    }
    private void loadImageFromDisk() {
        // load the image
        imageData = new byte[]{};
    }
    public void draw() {
        System.out.println("Called BrowserImage.draw()");
    }

}

// the Proxy class
class ProxyImage implements ISubject {

    private BrowserImage image = null;
    private String filename = null;
    // Constructor
    public ProxyImage(final String filename) {
        this.filename = filename;
    }
    public void draw() {
        if (image == null) {
            image = new BrowserImage(filename);
        }
        image.draw();
    }
}

public class ProxyExample {
    public static void main(final String[] arguments) {
        final ISubject image1 = new ProxyImage("test-image1.jpg");
        final ISubject image2 = new ProxyImage("test-image2.jpg");

        image1.draw(); // loading necessary
        image1.draw(); // loading unnecessary
        image2.draw(); // loading necessary
        image2.draw(); // loading unnecessary
        image1.draw(); // loading unnecessary
    }
}
