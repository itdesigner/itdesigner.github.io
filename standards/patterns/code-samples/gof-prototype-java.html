import java.util.Objects;
import java.util.ArrayList;
import java.util.List;

interface IEmployee {
    IEmployee clone();
    String getDetails();
}
// concrete prototype
class Developer implements IEmployee {
    public int WordsPerMinute;
    public String Name;
    public String Role;
    public String PreferredLanguage;

    public Developer() { }
    public Developer(Developer target) {
        if(target != null) {
        this.WordsPerMinute = target.WordsPerMinute;
        this.Name = target.Name;
        this.Role = target.Role;
        this.PreferredLanguage = target.PreferredLanguage;
        }
    }
    public IEmployee clone() {
        //here we might want to use a native clone method
        return new Developer(this);
    }
    public String getDetails()
    {
            return this.Name + " - " + this.Role + ": " + this.PreferredLanguage + " language";
    }
}
// concrete prototype
class Admin implements IEmployee {
    public int WordsPerMinute;
    public String Name;
    public String Role;

    public Admin() { }
    public Admin(Admin target) {
        if(target != null) {
        this.WordsPerMinute = target.WordsPerMinute;
        this.Name = target.Name;
        this.Role = target.Role;
        }
    }
    public IEmployee clone() {
        //here we might want to use a native clone method
        return new Admin(this);
    }
    public String getDetails()
    {
        return this.Name + " - " + this.Role + ": " + this.WordsPerMinute + "wpm";
    }
}
public class Demo {
    public static void main(String[] args) {
        List<IEmployee> employees = new ArrayList<>();

        Developer dev = new Developer();
        dev.Name = "Michael";
        dev.Role = "Team Lead";
        dev.PreferredLanguage = "Java";
        employees.add(dev);

        Developer anotherDev = (Developer) dev.clone();
        anotherDev.Name = "Jose";
        employees.add(anotherDev);

        Admin admin = new Admin();
        admin.Name = "Jane";
        admin.Role = "Administrative Assistant";
        admin.WordsPerMinute = 150;
        employees.add(admin);

        Admin anotherAdmin = (Admin) admin.clone();
        anotherAdmin.Name = "Robert";
        employees.add(anotherAdmin);
        
        for(int i=0; i < employees.size(); i++) {
            System.out.println(i + " " + employees.get(i).getDetails());
        }
    }
}
